391 				data : $('#login-form').serialize(),
392 				dataType: 'json',
393 				
394 				//communication success
395 				success:function(data){
396 					
397 					if(data.result){
398 						sessionStorage.login_token = data.token;
399 						$('#login-form').hide();
400 						$('#logout-form').show();
401 						$('#password').val("");
402 						$('#login-msg').attr('class','good').html("Login Success.").show();
403 					} else {

406 				},
407 				
408 				//communication error

1271 			var dLat = (lat2-lat1) * 3.14/180;
1272 			var dLon = (lon2-lon1) * 3.14/180;
